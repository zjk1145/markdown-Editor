#include<cstdio>

int a[10000];
int b[10000];
int n;
void dfs(int x)
{
    if(x==n+1)
    {
        for(int i=1;i<=n;i++)printf("%5d",a[i]);
        printf("\n");
    }
    else
    {
        for(int i=1;i<=n;i++)
        {
            if(b[i]==0)
            {
                b[i]=1;
                a[x]=i;
                dfs(x+1);
                b[i]=0;
                a[x]=0; 
            }
        }
    }
}
int main()
{
    scanf("%d",&n);
    memset(b,0,sizeof(b));
    dfs(1);
}
